454 subroutine get_timestep(current_iter, n_iter, current_time, tprint, dt, dump_flag)
455 
456  use parameters, only : nx, ny, nz, cfl, mpi_real_kind
457  use globals, only : primit, dx, dy, dz
458  implicit none
459 #ifdef MPIP
460  include "mpif.h"
461 #endif
462  integer, intent(in) :: current_iter, n_iter
463  real, intent(in) :: current_time, tprint
464  real, intent(out) :: dt
465  logical, intent(out) :: dump_flag
466  real :: dtp
467 #ifdef MHD
468  real :: cx, cy, cz
469 #else
470  real :: c
471 #endif
472  integer :: i, j, k, err
473 
474  dtp=1.e30
475 
476  do k=1,nz
477  do j=1,ny
478  do i=1,nx
479 
480 #ifdef MHD
481  call cfast(primit(5,i,j,k),primit(1,i,j,k),&
482  primit(6,i,j,k), primit(7,i,j,k), primit(8,i,j,k), &
483  cx,cy,cz)
484  dtp=min(dtp,dx/(abs(primit(2,i,j,k))+cx))
485  dtp=min(dtp,dy/(abs(primit(3,i,j,k))+cy))
486  dtp=min(dtp,dz/(abs(primit(4,i,j,k))+cz))
487 #else
488  call csound(primit(5,i,j,k),primit(1,i,j,k),c)
489  dtp=min(dtp,dx/(abs(primit(2,i,j,k))+c))
490  dtp=min(dtp,dy/(abs(primit(3,i,j,k))+c))
491  dtp=min(dtp,dz/(abs(primit(4,i,j,k))+c))
492 #endif
493  end do
494  end do
495  end do
496 
497  if (current_iter <= n_iter ) then
498  ! boots up simulation with small CFL
499  dtp = cfl*(2.**(-real(n_iter+1-current_iter)))*dtp
500  else
501  dtp=cfl*dtp
502  end if
503 
504 #ifdef MPIP
505  call mpi_allreduce(dtp, dt, 1, mpi_real_kind, mpi_min, mpi_comm_world,err)
506 #else
507  dt=dtp
508 #endif
509 
510  ! Adjust dt so t+dt doesnt overshoot tprint
511  if(( current_time + dt )>=tprint) then
512  dt= tprint-current_time
513  dump_flag=.true.
514  endif
515 
516 
517 end subroutine get_timestep

533 
534  implicit none
535  real, dimension(neq), intent(inout) :: pl, pr
536  real, dimension(neq), intent(in) :: pll, prr
537  integer, intent (in) :: neq

540

542  dl=pl(ieq)-pll(ieq)

544  dr=prr(ieq)-pr(ieq)

553  real function average(a,b)
554  implicit none
555  real, intent(in) :: a, b
556 
557 #if LIMITER==-1
558 
559  ! no average (reduces to 1st order)
560  average=0.
561 #endif
562 
563 #if LIMITER==0
564  ! no limiter
565  average=0.5*(a+b)
566 #endif
567 
568 #if LIMITER==1
569  ! Minmod - most diffusive
570  real :: s
571  s=sign(1.,a)
572  average=s*max(0.,min(abs(a),s*b))
573 #endif
574 
575 #if LIMITER==2
576  ! Falle Limiter (Van Leer)
577  if(a*b.le.0.) then
578  average=0.
579  else
580  average=a*b*(a+b)/(a**2+b**2)
581  end if
582 #endif
583 
584 #if LIMITER==3
585  ! Van Albada
586  real, parameter :: delta=1.e-7
587  average=(a*(b*b+delta)+b*(a*a+delta))/(a*a+b*b+delta)
588 #endif
589 
590 #if LIMITER==4
591  ! UMIST Limiter - less diffusive
592  real :: s, c, d
593  s=sign(1.,a)
594  c=0.25*a+0.75*b
595  d=0.75*a+0.25*b
596  average=min(2.*abs(a),2.*s*b,s*c,s*d)
597  average=s*max(0.,average)
598 #endif
599 
600 #if LIMITER==5
601  ! Woodward Limiter (o MC-limiter; monotonized central difference)
602  real :: s, c
603  s=sign(1.,a)
604  c=0.5*(a+b)
605  average=min(2.*abs(a),2.*s*b,s*c)
606  average=s*max(0.,average)
607 #endif
608 #if LIMITER==6
609  ! superbee Limiter (tends to flatten circular waves)
610  real :: s, av1, av2
611  s=sign(1.,b)
612  av1=min(2.*abs(b),s*a)
613  av2=min(abs(b),2.*s*a)
614  average=s*max(0.,av1,av2)
615 #endif
616 
617  end function average
618 
619 end subroutine limiter
