189 subroutine heatfluxes()
190  use hydro_core, only : csound
191  implicit none
192  integer :: i, j, k
193  real, parameter :: clight=3.e10, phi=0.3
194  real:: cs, coef, dTx, dTy, dTz, meanT, meanP, meanDens, yhp
195 
196  do k=0,nz
197  do j=0,ny
198  do i=0,nx
199 
200  yhp=1.!-primit(neqdyn+1,i,j,k)/primit(1,i,j,k)
201 
202  ! get the flux in the X direction
203  if (temp(i,j,k) == temp(i+1,j,k) ) then
204  f(5,i,j,k)=0.
205  else
206  meanp = 0.5*(primit(5,i,j,k)+primit(5,i+1,j,k))
207  meandens= 0.5*(primit(1,i,j,k)+primit(1,i+1,j,k))
208  meant = 0.5*( temp(i,j,k)+ temp(i+1,j,k))
209  dtx=(temp(i+1,j,k)-temp(i,j,k))/(dx*rsc)
210 
211  if (tc_saturation) then
212  call csound(meanp,meandens,cs)
213  cs=min(cs*sqrt(vsc2),clight)
214  coef=min( ksp(meant) , 5.*ph*cs*meanp*psc/abs(dtx) )
215  else
216  coef = ksp(meant)
217  end if
218 
219  f(5,i,j,k)=-coef*dtx*yhp
220 
221  end if
222  ! get the flux in the Y direction
223  if (temp(i,j,k) == temp(i,j+1,k) ) then
224  g(5,i,j,k)=0.
225  else
226  meanp = 0.5*(primit(5,i,j,k)+primit(5,i,j+1,k))
227  meandens= 0.5*(primit(1,i,j,k)+primit(1,i,j+1,k))
228  meant = 0.5*( temp(i,j,k)+ temp(i,j+1,k))
229  dty=(temp(i,j+1,k)-temp(i,j,k))/(dy*rsc)
230 
231  if (tc_saturation) then
232  call csound(meanp,meandens,cs)
233  cs=min(cs*sqrt(vsc2),clight)
234  coef=min( ksp(meant) , 5.*ph*cs*meanp*psc/abs(dty) )
235  else
236  coef = ksp(meant)
237  endif
238 
239  g(5,i,j,k)=-coef*dty*yhp
240 
241  end if
242  ! get the flux in the Z direction
243  if (temp(i,j,k) == temp(i,j,k+1) ) then
244  h(5,i,j,k)=0.
245  else
246  meanp = 0.5*(primit(5,i,j,k)+primit(5,i,j,k+1))
247  meandens= 0.5*(primit(1,i,j,k)+primit(1,i,j,k+1))
248  meant = 0.5*( temp(i,j,k) + temp(i,j,k+1))
249  dtz=(temp(i,j,k+1)-temp(i,j,k))/(dz*rsc)
250 
251  if (tc_saturation) then
252  call csound(meanp,meandens,cs)
253  cs=min(cs*sqrt(vsc2),clight)
254  coef=min( ksp(meant) , 5.*ph*cs*meanp*psc/abs(dtz) )
255  else
256  coef = ksp(meant)
257  endif
258 
259  h(5,i,j,k)=-coef*dtz*yhp
260 
261  end if
262 
263  end do
264  end do
265  end do

277 subroutine mhd_heatfluxes()

279  use hydro_core, only : csound
280  implicit none
281  integer :: i,j,k
282  real,parameter :: clight=3.e10,phi=0.3,alpha=5.0*phi
283  real :: cs,meanTemp,meanPres,meanDens
284  !Gradient Temperature
285  real :: gradTx,gradTy,gradTz
286  !real :: gradT
287  real :: gradT_parl_x,gradT_parl_y,gradT_parl_z,gradT_parl
288  real :: gradT_perp_x,gradT_perp_y,gradT_perp_z,gradT_perp
289  !Magnetic fiel unitary vector components
290  real :: bx,by,bz,modB,B2
291  !Internal product of vectors: B and gradT
292  real :: bgradT
293  !real :: ,bgradTx,bgradTy,bgradTz
294  !Coeficientes de conducción
295  real :: K_parl_x,K_parl_y,K_parl_z,K_perp_x,K_perp_y,K_perp_z
296  real :: coefSatx,coefSaty,coefSatz
297 
298  f(5,:,:,:)=0.0 ; g(5,:,:,:)=0.0 ; h(5,:,:,:)=0.0
299 
300  do k=0,nz
301  do j=0,ny
302  do i=0,nx
303 
304  bx = primit(6,i,j,k)
305  by = primit(7,i,j,k)
306  bz = primit(8,i,j,k)
307  b2 = bx*bx+by*by+bz*bz
308  modb = sqrt(b2)
309  bx = bx/modb
310  by = by/modb
311  bz = bz/modb
312 
313  ! not saturated conduction
314  if(.not.tc_saturation) then
315  ! get the flux in the X direction
316  if ( abs(temp(i,j,k)-temp(i+1,j,k)) < 1.0e-14 ) then
317 
318  gradtx = 0.0
319  k_parl_x = 0.0
320  k_perp_x = 0.0
321 
322  else
323 
324  meandens = 0.5*(primit(1,i,j,k)+primit(1,i+1,j,k))
325  meantemp = 0.5*( temp(i,j,k)+ temp(i+1,j,k))
326 
327  gradtx = (temp(i+1,j,k)-temp(i,j,k))/(dx*rsc)
328  k_parl_x = ksp_parl(meantemp)
329  k_perp_x = ksp_perp(meantemp,meandens*rhosc,b2*bsc**2)
330 
331  end if
332 
333  ! get the flux in the Y direction
334  if ( abs(temp(i,j,k)-temp(i,j+1,k)) < 1.0e-14 ) then
335 
336  gradty = 0.0
337  k_parl_y = 0.0
338  k_perp_y = 0.0
339 
340  else
341 
342  meandens = 0.5*(primit(1,i,j,k)+primit(1,i,j+1,k))
343  meantemp = 0.5*( temp(i,j,k)+ temp(i,j+1,k))
344 
345  gradty = (temp(i,j+1,k)-temp(i,j,k))/(dy*rsc)
346  k_parl_y = ksp_parl(meantemp)
347  k_perp_y = ksp_perp(meantemp,meandens*rhosc,b2*bsc**2)
348 
349  end if
350 
351  ! get the flux in the Z direction
352  if ( abs(temp(i,j,k)-temp(i,j,k+1)) < 1.0e-14 ) then
353 
354  gradtz = 0.0
355  k_parl_z = 0.0
356  k_perp_z = 0.0
357 
358  else
359 
360  meandens = 0.5*(primit(1,i,j,k)+primit(1,i,j,k+1))
361  meantemp = 0.5*( temp(i,j,k) + temp(i,j,k+1))
362 
363  gradtz = (temp(i,j,k+1)-temp(i,j,k))/(dz*rsc)
364  k_parl_z = ksp_parl(meantemp)
365  k_perp_z = ksp_perp(meantemp,meandens*rhosc,b2*bsc**2)
366 
367  end if
368 
369  !internal product of b.gradT
370  bgradt = bx*gradtx+by*gradty+bz*gradtz
371 
372  gradt_parl_x = bgradt*bx
373  gradt_parl_y = bgradt*by
374  gradt_parl_z = bgradt*bz
375 
376  gradt_perp_x = gradtx-gradt_parl_x
377  gradt_perp_y = gradty-gradt_parl_y
378  gradt_perp_z = gradtz-gradt_parl_z
379 
380  f(5,i,j,k) = -k_parl_x*gradt_parl_x - k_perp_x*gradt_perp_x
381 
382  g(5,i,j,k) = -k_parl_y*gradt_parl_y - k_perp_y*gradt_perp_y
383 
384  h(5,i,j,k) = -k_parl_z*gradt_parl_z - k_perp_z*gradt_perp_z
385 
386  else
387  ! Saturated conduction
388  ! get the flux in the X direction
389  if ( abs(temp(i,j,k)-temp(i+1,j,k)) < 1.0e-14 ) then
390 
391  gradtx = 0.0
392  k_parl_x = 0.0
393  k_perp_x = 0.0
394  coefsatx = 0.0
395 
396  else
397 
398  meanpres = 0.5*(primit(5,i,j,k)+primit(5,i+1,j,k))
399  meandens = 0.5*(primit(1,i,j,k)+primit(1,i+1,j,k))
400  meantemp = 0.5*( temp(i,j,k)+ temp(i+1,j,k))
401  call csound(meanpres,meandens,cs)
402  cs = min(cs*vsc,clight)
403  coefsatx = alpha*meandens*cs**3
404 
405  gradtx = (temp(i+1,j,k)-temp(i,j,k))/(dx*rsc)
406  k_parl_x = ksp_parl(meantemp)
407  k_perp_x = ksp_perp(meantemp,meandens*rhosc,b2*bsc**2)
408 
409  end if
410 
411  ! get the flux in the Y direction
412  if ( abs(temp(i,j,k)-temp(i,j+1,k)) < 1.0e-14 ) then
413 
414  gradty = 0.0
415  k_parl_y = 0.0
416  k_perp_y = 0.0
417  coefsaty = 0.0
418 
419  else
420 
421  meanpres = 0.5*(primit(5,i,j,k)+primit(5,i,j+1,k))
422  meandens = 0.5*(primit(1,i,j,k)+primit(1,i,j+1,k))
423  meantemp = 0.5*( temp(i,j,k)+ temp(i,j+1,k))
424  call csound(meanpres,meandens,cs)
425  cs = min(cs*vsc,clight)
426  coefsaty = alpha*meandens*cs**3
427 
428  gradty = (temp(i,j+1,k)-temp(i,j,k))/(dy*rsc)
429  k_parl_y = ksp_parl(meantemp)
430  k_perp_y = ksp_perp(meantemp,meandens*rhosc,b2*bsc**2)
431 
432  end if
433 
434  ! get the flux in the Z direction
435  if ( abs(temp(i,j,k)-temp(i,j,k+1)) < 1.0e-14 ) then
436 
437  gradtz = 0.0
438  k_parl_z = 0.0
439  k_perp_z = 0.0
440  coefsatz = 0.0
441 
442  else
443 
444  meanpres = 0.5*(primit(5,i,j,k)+primit(5,i,j,k+1))
445  meandens = 0.5*(primit(1,i,j,k)+primit(1,i,j,k+1))
446  meantemp = 0.5*( temp(i,j,k) + temp(i,j,k+1))
447  call csound(meanpres,meandens,cs)
448  cs = min(cs*vsc,clight)
449  coefsatz = alpha*meandens*cs**3
450 
451  gradtz = (temp(i,j,k+1)-temp(i,j,k))/(dz*rsc)
452  k_parl_z = ksp_parl(meantemp)
453  k_perp_z = ksp_perp(meantemp,meandens*rhosc,b2*bsc**2)
454 
455  end if
456 
457  !internal product of b.gradT
458  bgradt = bx*gradtx+by*gradty+bz*gradtz
459 
460  gradt_parl_x = bgradt*bx
461  gradt_parl_y = bgradt*by
462  gradt_parl_z = bgradt*bz
463  ! |gradT_parl| == |(b.gradT)b| == |(b.gradT)|
464  gradt_parl = bgradt
465 
466  gradt_perp_x = gradtx-gradt_parl_x
467  gradt_perp_y = gradty-gradt_parl_y
468  gradt_perp_z = gradtz-gradt_parl_z
469  ! |gradT_perp| == |gradT-gradT_parl|
470  gradt_perp = sqrt( gradt_perp_x*gradt_perp_x + gradt_perp_y*gradt_perp_y + gradt_perp_z*gradt_perp_z )
471 
472  f(5,i,j,k) = - 1./( 1./(k_parl_x + 1.e-14) + gradt_parl/(coefsatx + 1.e-14) ) * gradt_parl_x &
473  - 1./( 1./(k_perp_x + 1.e-14) + gradt_perp/(coefsatx + 1.e-14) ) * gradt_perp_x
474 
475  g(5,i,j,k) = - 1./( 1./(k_parl_y + 1.e-14) + gradt_parl/(coefsaty + 1.e-14) ) * gradt_parl_y &
476  - 1./( 1./(k_perp_y + 1.e-14) + gradt_perp/(coefsaty + 1.e-14) ) * gradt_perp_y
477 
478  h(5,i,j,k) = - 1./( 1./(k_parl_z + 1.e-14) + gradt_parl/(coefsatz + 1.e-14) ) * gradt_parl_z &
479  - 1./( 1./(k_perp_z + 1.e-14) + gradt_perp/(coefsatz + 1.e-14) ) * gradt_perp_z
480 
481  end if
482 
483  end do
484  end do
485  end do

487 end subroutine mhd_heatfluxes

540 
541  call mpi_sendrecv(sendo, bzsize, mpi_real_kind, out , 0, &
542  recvi, bzsize, mpi_real_kind, in , 0, &
543  comm3d, status , err)

546  if (right .ne. -1) u(5,nxp1 ,0:nyp1,0:nzp1)=recvr(1,1,:,:)
