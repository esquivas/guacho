80 subroutine grav_source(xc,yc,zc,pp,s)
81  use constants, only : ggrav
82  use exoplanet ! this module contains the position of the planet
83  implicit none
84  real, intent(in) :: xc, yc, zc
85  real, intent(in) :: pp(neq)
86  real, intent(inout) :: s(neq)
87  integer, parameter :: nb=2 ! 2 particles
88  real :: x(nb),y(nb),z(nb), GM(nb), rad2(nb)
89  integer :: i
90 
91  gm(1)=0.3*ggrav*masss/rsc/vsc2
92  gm(2)=ggrav*massp/rsc/vsc2
93 
94  !calculate distance from the sources
95  ! star
96  x(1)=xc
97  y(1)=yc
98  z(1)=zc
99  rad2(1) = x(1)**2 +y(1)**2 + z(1)**2
100  ! planet
101  x(2)=xc-xp
102  y(2)=yc
103  z(2)=zc-zp
104  rad2(2) = x(2)**2 +y(2)**2 + z(2)**2
105 
106  ! update source terms
107  do i=1, nb
108  ! momenta
109  s(2)= s(2)-pp(1)*gm(i)*x(i)/(rad2(i)**1.5)
110  s(3)= s(3)-pp(1)*gm(i)*y(i)/(rad2(i)**1.5)
111  s(4)= s(4)-pp(1)*gm(i)*z(i)/(rad2(i)**1.5)
112  ! energy
113  s(5)= s(5)-pp(1)*gm(i)*( pp(2)*x(i) +pp(3)*y(i) +pp(4)*z(i) ) &
114  /(rad2(i)**1.5 )
115  end do
116 
117 
118 end subroutine grav_source
