319  integer :: i,j,k, iobs, jobs
320  real :: x,y,z,xn,yn,zn
321 
322  do k=1,nz

343  map(iobs, jobs) = map(iobs, jobs) + u(1,i,j,k)*rhosc*dz*rsc

346  end if
