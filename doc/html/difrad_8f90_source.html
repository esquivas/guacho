146 subroutine random_versor(xd,yd,zd)
147 
148  implicit none
149  real, intent (out) :: xd, yd, zd
150  real :: r2, w
151  real :: ran(3)

155  call random_number(ran)
156  xd=2.*(ran(1)-0.5)
157  yd=2.*(ran(2)-0.5)
158  zd=2.*(ran(3)-0.5)

184 subroutine starsource(srad,x0,y0,z0,x,y,z,xd,yd,zd)
185  implicit none
186  real, intent (in) :: Srad, x0, y0, z0
187  real, intent (out) :: x, y, z, xd, yd, zd
188  real :: xs, ys,zs
189  real :: rd, rs, w, prod
190  real :: ran(3)
191 
192  ! get unit vector from star center
193  rs=2.
194  do while(rs > 1.)
195  call random_number(ran)
196  xs=2.*(ran(1)-0.5)
197  ys=2.*(ran(2)-0.5)
198  zs=2.*(ran(3)-0.5)
199  rs= xs**2 + ys**2 + zs**2
200  end do
201  w=srad/sqrt(rs)
202  xs=xs*w
203  ys=ys*w
204  zs=zs*w
205 
206  ! the position of the source
207  x=x0+xs
208  y=y0+ys
209  z=z0+zs
210 
211  ! get the direction of the ray
212  rd=2.
213  do while(rd > 1.)
214  call random_number(ran)
215  xd=2.*(ran(1)-0.5)
216  yd=2.*(ran(2)-0.5)
217  zd=2.*(ran(3)-0.5)
218  rd = xd**2 + yd**2 + zd**2
219  end do
220  w=1./sqrt(rd)
221  xd=xd*w
222  yd=yd*w
223  zd=zd*w
224 
225  ! reflect rays that point to the star
226  prod=( xs*xd+ ys*yd + zs*zd )
227  if (prod < 0. ) then
228  xd=-xd
229  yd=-yd
230  zd=-zd
231  endif
232 
233 end subroutine starsource
