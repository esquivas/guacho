57 #ifdef MPIP

59  write(file2,'(a)') trim(outputpath)//'VTK/master.visit'

71  end do
72  close(7)
73  end if
74  end if
75 
76  write(file1,'(a,i3.3,a,i3.3,a)') trim(outputpath)//'VTK/out-',rank,'.',itprint,'.vtk'
77  unitout=rank+10
78 #else

80  unitout=10
81 #endif
82 
83  open(unit=unitout,file=file1,status='replace',access='stream')!, convert='BIG_ENDIAN')
84 
85  ! write the header
86  x0=( real(coords(0)*nx) )*dx
87  y0=( real(coords(1)*ny) )*dy
88  z0=( real(coords(2)*nz) )*dz

99

102

104  write(unitout) trim(cbuffer),lf

111 
112  ! writes the variables, scalars first then vectors
113

116 
117  ! DENSITY
118  write(cbuffer,'(a)') 'FIELD FieldData 1'

146 
147  ! TEMP
148  write(cbuffer,'(a)') 'FIELD FieldData 1'

150  write(cbuffer,'(a,1x,i0,1x,i0,a)') 'Temperature',1,ncells,'float'

162 
163  ! VELOCITY
164  write(cbuffer,'(a)') 'VECTORS Velocity float'
165  write(unitout) trim(cbuffer),lf
166  do k=1,nz
167  do j=1,ny
168  do i=1,nx
169  write(unitout) real(primit(2,i,j,k)*vsc,4), &
170  real(primit(3,i,j,k)*vsc,4), &
171  real(primit(4,i,j,k)*vsc,4)
172  end do
173  end do
174  end do
175  write(unitout) lf
176 
177  if (pmhd .or. mhd) then
178  ! MAGNETIC FIELD
179  write(cbuffer,'(a)') 'VECTORS BField float'
180  write(unitout) trim(cbuffer),lf
181  do k=1,nz
182  do j=1,ny
183  do i=1,nx
184  write(unitout) real(primit(6,i,j,k)*bsc,4), &
185  real(primit(7,i,j,k)*bsc,4), &
186  real(primit(8,i,j,k)*bsc,4)
187  end do
188  end do
189  end do
190  write(unitout) lf
191  end if
192 
193  close(unitout)
194 
195  print'(i3,a,a)',rank," wrote file:",trim(file1)
