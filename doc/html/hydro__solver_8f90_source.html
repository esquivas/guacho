84  real :: s(neq)
85  real, intent(in) :: dt

88 
89  dtdx=dt/dx
90  dtdy=dt/dy

93  if (enable_flux_cd) call get_current()

95  do k=1,nz
96  do j=1,ny

98 
99  if (.not.enable_flux_cd) then
100  ! upwind step for all variables
101  up(:,i,j,k)=u(:,i,j,k)-dtdx*(f(:,i,j,k)-f(:,i-1,j,k)) &
102  -dtdy*(g(:,i,j,k)-g(:,i,j-1,k)) &
103  -dtdz*(h(:,i,j,k)-h(:,i,j,k-1))
104  else
105 
106  call flux_cd_update(i,j,k,dt)
107 
108  endif
109 
110  if (user_source_terms .or. &
111  radiation_pressure .or. &
112  eight_wave) then
113  call source(i,j,k,primit(:,i,j,k),s)
114  up(:,i,j,k)= up(:,i,j,k)+dt*s(:)
115 
116  end if

118  end do

129 subroutine tstep()
130 
131  use parameters, only : tsc, riemann_solver, eq_of_state, &
132  dif_rad, cooling, &
133  th_cond
134  use constants
135  use globals
136  use hydro_core, only : calcprim
137  use boundaries
138  use cooling_h
139  use cooling_dmc
140  use cooling_chi
141  use difrad
142  use thermal_cond
143  implicit none
144  real :: dtm
145 
146  ! 1st half timestep ========================
147  dtm=dt_cfl/2.
148  ! calculate the fluxes using the primitives
149  ! (piecewise constant)
150  if (riemann_solver == solver_hll ) call hllfluxes(1)
151  if (riemann_solver == solver_hllc) call hllcfluxes(1)
152  if (riemann_solver == solver_hlle) call hllefluxes(1)
153  if (riemann_solver == solver_hlld) call hlldfluxes(1)
154  !if (riemann_solver == SOLVER_HLLE_SPLIT_B) call hllefluxes(1)
155  !if (riemann_solver == SOLVER_HLLD_SPLIT_B) call hllefluxes(1)
156  if (riemann_solver == solver_hlle_split_all) call hllefluxessplitall(1)
157  !if (riemann_solver == SOLVER_HLLD_SPLIT_ALL) call hllefluxes(1)
158 
159 
160  ! upwind timestep
161  call step(dtm)
162 
163  ! add viscosity
164  !call viscosity()
165 
166  ! 2nd half timestep ========================
167  ! boundaries in up and primitives up ---> primit
168  call boundaryii()
169  call calcprim(up,primit)
170 
171  ! calculate the fluxes using the primitives
172  ! with linear reconstruction (piecewise linear)
173  if (riemann_solver == solver_hll ) call hllfluxes(2)
174  if (riemann_solver == solver_hllc) call hllcfluxes(2)
175  if (riemann_solver == solver_hlle) call hllefluxes(2)
176  if (riemann_solver == solver_hlld) call hlldfluxes(2)
177  !if (riemann_solver == SOLVER_HLLE_SPLIT_B) call hllefluxes(2)
178  !if (riemann_solver == SOLVER_HLLD_SPLIT_B) call hllefluxes(2)
179  if (riemann_solver == solver_hlle_split_all) call hllefluxessplitall(2)
180  !if (riemann_solver == SOLVER_HLLD_SPLIT_ALL) call hllefluxes(2)
181 
182  ! upwind timestep
183  call step(dt_cfl)
184 
185  ! add viscosity
186  call viscosity()
187 
188  ! copy the up's on the u's
189  u=up
190 
191  ! update the chemistry network
192  ! at this point is in cgs
193  ! the primitives in the physical cell are upated
194  if (eq_of_state == eos_chem) call update_chem()
195 
196  ! Do the Radiation transfer (Monte Carlo type)
197  if (dif_rad) call diffuse_rad()
198 
199  !-------------------------
200  ! apply cooling/heating terms
201 
202  ! add cooling (H rat e)to the conserved variables
203  if (cooling == cool_h) then
204  call coolingh()
205  ! update the primitives with u
206  call calcprim(u, primit)
207  end if
208 
209  ! DMC cooling (the primitives are updated in the cooling routine)
210  if (cooling == cool_dmc) call coolingdmc()
211 
212  ! Chianti cooling (the primitives are updated in the cooling routine)
213  if (cooling == cool_chi) call coolingchi()
214 
215  ! Chemistry network cooling (primitives are already updated in update_chem)
216  if (cooling == cool_chem) call cooling_chem()
217 
218  ! boundary contiditions on u
219  call boundaryi()
220 
221  ! update primitives on the boundaries
222  call calcprim(u,primit,only_ghost=.true.)
223 
224  ! Thermal conduction
225  if (th_cond /= 0 ) call thermal_conduction()
226 
227 end subroutine tstep
